#include <iostream.h>
      
class Account
 {  
   public:  
   float salary;
      Account() {
    salary  = 60000;
      }
 };  
   class Programmer: public Account 
   {  
   public:  
   float bonus; 
      Programmer() {
      bonus=5000;
      }
   };       
int main() 
{  
     Programmer p1;  
     cout<<"Salary: "<<p1.salary<<endl;    
     cout<<"Bonus: "<<p1.bonus<<endl;    
    return 0;  
}
